INFLUENCIA DE VALORES ATÍPICOS (CON DISTANCIA DE COOK)

# Obtenemos medidas de diagnóstico
model_data <- augment(modelo) %>%
  mutate(index = 1:n())

# Gráfico de Distancia de Cook
ggplot(model_data, aes(index, .cooksd)) +
  geom_point(aes(color = .cooksd > (4/nrow(data))), alpha = 0.5) +
  geom_hline(yintercept = 4/nrow(data), linetype = "dashed", color = "red") +
  labs(title = "Distancia de Cook para identificar influyentes",
       x = "Índice de observación",
       y = "Distancia de Cook") +
  theme_minimal()
